                    \MT@detokenize@n{#1}-\MT@detokenize@n{#2}\endcsname}%

    \fi

        \edef\MT@char{\MT@exp@two@c\MT@strip@prefix\meaning\MT@char>\relax}%

            \MT@char\MT@charxstring\relax\relax\relax\relax\relax
      \fi
    \fi
  \fi
